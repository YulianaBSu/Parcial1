#include <iostream>
using namespace std;

int size = 8;
int main() {
    //PATRON 3
    char matriz[8][8];

    // Inic la matriz con espacios en blanco
    for(int i = 0; i < size; i++) {
        for(int j = 0; j < size; j++) {
            matriz[i][j] = ' ';
        }
    }
    for (int i = 0; i < size; i++) {
        for (int j = 0; j < size; j++) {
            // PatrÃ³n de marcado
            if (
                (i == 0 || i == 1 || i == 4 || i == 5) &&
                (j == 0 || j == 1 || j == 3 || j == 4 || j == 6 || j == 7)
            ) {
                matriz[i][j] = 'X';
            }
            else if (
                (i == 2 || i == 3 || i == 6 || i == 7) &&
                (j == 1 || j == 2 || j == 4 || j == 5 || j == 7)
            ) {
                matriz[i][j] = 'X';
            }
        }
    }

    // Imprimir la matriz
    for (int i = 0; i < size; i++) {
        for (int j = 0; j < size; j++) {
            cout << matriz[i][j] << " ";
        }
        cout << endl;
    }

    return 0;
}
