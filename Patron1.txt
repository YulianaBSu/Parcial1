#include <iostream>
using namespace std;

const int filas = 8;
const int columnas = 8;

int main() {

//Definir arreglo de punteros
    int **matriz = new int*[filas];
    for (int i = 0; i < filas; i++) {
	matriz[i] = new int[columnas]();
    }

    for (int i = 0; i < filas; i++) {
        for (int j = i; j < columnas - i; j++) {
            matriz[i+4][j] = 1;
            matriz[filas-i-5][j] = 1;
        }
    }

    // Imprimir la matriz
    for (int i = 0; i < filas; i++) {
        for (int j = 0; j < columnas; j++) {
            cout << matriz[i][j];
        }
        cout << endl;
    }

    return 0;
}
