#include <iostream>
using namespace std;

const int size = 8;
int main() {
    //PATRON 2
    char matriz[8][8];

    // Inic la matriz con espacios en blanco
    for(int i = 0; i < size; i++) {
        for(int j = 0; j < size; j++) {
            matriz[i][j] = ' ';
        }
    }

    // Patron con X
    for(int i = 0; i < size; i++) {
        matriz[i][i] = 'X';
        matriz[i][size - i - 1] = 'X';
    }

    // Imprimir la matriz
    for(int i = 0; i < size; i++) {
        for(int j = 0; j < size; j++) {
            cout<<matriz[i][j] << " ";
        }
        cout<<endl;
    }
    return 0;
}
