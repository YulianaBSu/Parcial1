#include <iostream>
using namespace std;

const int filas = 8;
const int columnas = 8;

int main() {
    char matriz[filas][columnas];

    // Inic matriz con espacios en blanco
    for (int i = 0; i < filas; i++) {
        for (int j = 0; j < columnas; j++) {
            matriz[i][j] = ' ';
        }
    }

    // Llenar 4 espacios con "X" y desplazarse un espacio en cada iteraciÃ³n
    for (int i = 0; i < 4; i++) {
        for (int j = 0; j < 4; j++) {
            matriz[i][i+j] = 'X';
        }
        }

    // Revertir el ciclo a partir de la quinta fila
    for (int i = 4; i < filas; i++) {
        for (int j = 0; j < 4; j++) {
            matriz[i][(j - i + columnas-1) ] = 'X';
        }
    }

    // Imprimir la matriz
    for (int i = 0; i < filas; i++) {
        for (int j = 0; j < columnas; j++) {
            cout << matriz[i][j] << " ";
        }
        cout << endl;
    }

    return 0;
}
